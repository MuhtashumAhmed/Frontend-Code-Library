--------- Usage -------------

export const loginUser = async (credentials: LoginCredentials): Promise<LoginSuccessResponse> => {
    const response = await apiFetch<LoginResponse>(ENDPOINTS.AUTH.LOGIN, {
        method: "POST",
        body: credentials,
    });

    //  Type guard - check if success
    if (response.error) {
        throw new Error(response.message); // "invalid_credentials"
    }

    return response as LoginSuccessResponse; // Now data exists!
};
